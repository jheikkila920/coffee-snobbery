**A. `app/static/js/alpine-components/recipe-step-builder.js`** — first live Alpine CSP-build component.

Follow the convention from `app/static/js/alpine-components/__init.js` (registered via `Alpine.data('name', factory)` inside an `alpine:init` listener).

File header comment: "Recipe step builder — D-09 (zero round-trips during editing). CSP-build compliant: registered via Alpine.data; template uses x-data='recipeStepBuilder' (string reference). Loaded after the alpine-csp core script."

Implementation:

```
document.addEventListener('alpine:init', () => {
  Alpine.data('recipeStepBuilder', () => ({
    steps: [],
    init() {
      const initial = this.$root.dataset.initialSteps;
      try { this.steps = JSON.parse(initial || '[]'); } catch (_) { this.steps = []; }
      if (this.steps.length === 0) {
        this.steps = [{ water_grams: 50, time_seconds: 30, label: 'Bloom' }];
      }
    },
    addStep() {
      const prev = this.steps[this.steps.length - 1] || { water_grams: 0, time_seconds: 0, label: '' };
      this.steps.push({ water_grams: (prev.water_grams || 0) + 50, time_seconds: (prev.time_seconds || 0) + 45, label: '' });
    },
    removeStep(i) { this.steps.splice(i, 1); },
    moveUp(i) { if (i > 0) { const t = this.steps[i-1]; this.steps[i-1] = this.steps[i]; this.steps[i] = t; } },
    moveDown(i) { if (i < this.steps.length - 1) { const t = this.steps[i+1]; this.steps[i+1] = this.steps[i]; this.steps[i] = t; } },
    deltaWater(i) { return i === 0 ? this.steps[0].water_grams : this.steps[i].water_grams - this.steps[i-1].water_grams; },
    deltaTime(i) { return i === 0 ? this.steps[0].time_seconds : this.steps[i].time_seconds - this.steps[i-1].time_seconds; },
    get totalWater() { return this.steps.length ? this.steps[this.steps.length - 1].water_grams : 0; },
    get totalTime() { return this.steps.length ? this.steps[this.steps.length - 1].time_seconds : 0; },
    get stepsJson() { return JSON.stringify(this.steps); },
    get timelineSegments() {
      const total = Math.max(1, this.totalTime);
      let acc = 0;
      return this.steps.map((s) => {
        const delta = Math.max(0, s.time_seconds - acc);
        acc = s.time_seconds;
        return { label: s.label || '—', water: s.water_grams, time: s.time_seconds, ratio: (delta / total) * 100 };
      });
    },
  }));
});
```
