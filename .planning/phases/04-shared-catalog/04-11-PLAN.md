**A. `app/static/js/alpine-components/mini-modal.js`** — Alpine `miniModal` component.

File header comment: "Mini-modal Alpine component (D-15). Open/close + ESC + backdrop-click + dirty-check + HX-Trigger-driven close-on-success. CSP-build compliant: registered via Alpine.data."

Implementation:

```
document.addEventListener('alpine:init', () => {
  Alpine.data('miniModal', () => ({
    open: false,
    dirty: false,
    init() {
      // ESC key closes the modal (UI-SPEC §Mini-Modal).
      this._onKeydown = (e) => { if (e.key === 'Escape' && this.open) this.close(); };
      window.addEventListener('keydown', this._onKeydown);
    },
    destroy() { window.removeEventListener('keydown', this._onKeydown); },
    show() { this.open = true; this.dirty = false; },
    close() {
      if (this.dirty && !window.confirm('Discard unsaved changes?')) return;
      this.open = false;
      this.dirty = false;
      // Empty the mount so the next show() starts fresh.
      const mount = document.getElementById('modal-mount');
      if (mount) mount.innerHTML = '';
    },
    markDirty() { this.dirty = true; },
    onBackdropClick(e) { if (e.target === e.currentTarget) this.close(); },
  }));
});

// Global HTMX-event consumer: when the server emits HX-Trigger: {entity}-created,
// the parent coffee form's autocomplete listener handles pre-select (autocomplete.js);
// here we close the modal.
document.body.addEventListener('roaster-created', () => {
  const modal = document.getElementById('modal-mount');
  if (modal && modal._alpineState) modal._alpineState.open = false;
  // Fallback: just clear the mount innerHTML.
  if (modal) modal.innerHTML = '';
});
document.body.addEventListener('flavor-note-created', () => {
  const modal = document.getElementById('modal-mount');
  if (modal) modal.innerHTML = '';
});
```

LOCKED MECHANISM (N1): modal close == `document.getElementById('modal-mount').innerHTML = ''` from the global `roaster-created` / `flavor-note-created` listeners (see the two `document.body.addEventListener(...)` lines above). DO NOT depend on the fragile `modal._alpineState.open = false` access — Alpine does not expose component state via a stable property. The empty-mount-from-global-listener approach is the simpler, deterministic path: server emits HX-Trigger → HTMX dispatches the CustomEvent on `document.body` → the listener empties `#modal-mount`. The Alpine component itself never has to "know" the entity was created — its DOM is just removed.

**B. `app/static/js/alpine-components/autocomplete.js`** — Alpine `autocomplete` component.

File header comment: "Autocomplete Alpine component (D-13 + D-14 + D-16). Keyboard nav + selection commit + + Create new dispatch + HX-Trigger consumer for parent-form pre-select. CSP-build compliant."

Implementation:

```
document.addEventListener('alpine:init', () => {
  Alpine.data('autocomplete', (config) => ({
    // config = { entityKey: 'roaster' | 'flavor-note', hiddenInputName: 'roaster_id', initialId: 12, initialName: 'Onyx' }
    entityKey: '',
    hiddenInputName: '',
    selectedId: null,
    selectedLabel: '',
    query: '',
    open: false,
    highlightIdx: -1,

    init() {
      this.entityKey = (config && config.entityKey) || '';
      this.hiddenInputName = (config && config.hiddenInputName) || '';
      this.selectedId = (config && config.initialId) || null;
      this.selectedLabel = (config && config.initialName) || '';
      this.query = this.selectedLabel;

      // D-16: HX-Trigger from the server-created entity → pre-select.
      const eventName = `${this.entityKey}-created`;
      this._onCreated = (evt) => {
        const idKey = `${this.entityKey.replace('-', '_')}_id`;  // 'roaster' → 'roaster_id', 'flavor-note' → 'flavor_note_id'
        this.selectedId = evt.detail[idKey];
        this.selectedLabel = evt.detail.name;
        this.query = evt.detail.name;
        this.open = false;
      };
      document.body.addEventListener(eventName, this._onCreated);
    },

    destroy() {
      const eventName = `${this.entityKey}-created`;
      document.body.removeEventListener(eventName, this._onCreated);
    },

    onInput($el) {
      this.query = $el.value;
      this.open = this.query.length >= 2;
      // HTMX handles the actual debounce + fetch via hx-trigger on the input element.
    },

    onFocus() { this.open = this.query.length >= 2; },

    onBlur() { setTimeout(() => { this.open = false; }, 150); },  // delay so click on option fires first

    select(id, label) { this.selectedId = id; this.selectedLabel = label; this.query = label; this.open = false; },

    onKeydown(e) {
      // Read the visible options from the DOM (HTMX populated them).
      const items = e.currentTarget.parentElement.querySelectorAll('[role="option"]');
      if (e.key === 'ArrowDown') { e.preventDefault(); this.highlightIdx = Math.min(this.highlightIdx + 1, items.length - 1); }
      else if (e.key === 'ArrowUp') { e.preventDefault(); this.highlightIdx = Math.max(this.highlightIdx - 1, 0); }
      else if (e.key === 'Enter') { e.preventDefault(); const item = items[this.highlightIdx]; if (item) item.click(); }
      else if (e.key === 'Escape') { this.open = false; }
    },
  }));
});
```

This API is comprehensive. The CSP-build constraint: NO inline expression strings in templates beyond declarative bindings. The `onKeydown($event)` etc. references are method names on the component — allowed.
