    ```yaml
    # docker-compose.override.yml.example
    #
    # Copy this file to docker-compose.override.yml (gitignored) for the dev loop.
    # Compose auto-merges docker-compose.override.yml — no flags needed.
    #
    # With the override in place, `docker compose up` rebuilds the coffee-snobbery
    # image locally instead of pulling from GHCR, and `docker compose run --rm
    # coffee-snobbery-test` runs the full pytest gate using the `dev` Dockerfile
    # stage (which carries pytest, playwright, and the Chromium browser binary).
    #
    # Operators DO NOT need this file. It is dev-only.

    services:
      coffee-snobbery:
        build:
          context: .
          dockerfile: Dockerfile
          # Pin the prod image to the runtime stage. Without this, an omitted target
          # builds the LAST Dockerfile stage (`dev`), whose ENTRYPOINT is pytest — so
          # `up -d` runs the test suite and restart-loops instead of serving uvicorn.
          target: runtime

      coffee-snobbery-test:
        # Test runner — gated behind the `test` profile so `docker compose up -d`
        # never starts it. One-command gate:
        #   docker compose build coffee-snobbery-test && docker compose run --rm coffee-snobbery-test
        # Uses the `dev` Dockerfile stage which carries pytest, playwright,
        # and the Chromium binary (T-12-10 / T-12-11).
        build:
          context: .
          dockerfile: Dockerfile
          target: dev
        profiles: [test]
        depends_on:
          coffee-snobbery-db:
            condition: service_healthy
        env_file:
          - .env
        environment:
          # Same inline URL as the prod service; conftest creates a sibling *_test DB
          # and only ever touches that (T-12-11 / SNOB_CI interlock).
          DATABASE_URL: postgresql+psycopg://${POSTGRES_USER}:${POSTGRES_PASSWORD}@coffee-snobbery-db:5432/${POSTGRES_DB}
          SNOB_CI: "1"
        networks:
          - coffee-snobbery-net
        # No source bind-mount on purpose — the test gate runs the BAKED tree.
        # Refresh code with `docker compose build coffee-snobbery-test` before
        # running the suite, or use `docker compose cp` for fast per-file iteration
        # against the live coffee-snobbery container (see CONTRIBUTING.md).
    ```

    - The `coffee-snobbery` override's build `target:` is `runtime`, NOT `dev`.
    - LF line endings (no CRLF) — per `Windows-CRLF-on-pathlib-write-text` memory.
