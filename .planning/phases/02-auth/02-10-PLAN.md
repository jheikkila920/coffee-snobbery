The test walks the locked Phase 2 success-criterion flow:

1. GET /setup → 200 + form rendered (with csrftoken cookie set on response).
2. POST /setup with valid form + CSRF pair → 303 → /.
3. Confirm Set-Cookie session_id on the 303 response.
4. Follow to GET / with the session cookie → 200 + body contains "Signed in as &lt;username&gt;".
5. POST /logout with CSRF pair → 303 → /login.
6. Confirm session_id cookie is cleared (Max-Age=0).
7. GET / one more time (no session cookie) → 200 + body contains "Sign in" link.

```
"""Phase 2 cold-container E2E smoke test.

Asserts the ROADMAP success criterion (amended per D-01/D-03):
    cold container → /setup → auto-login → see / with "Signed in as <username>"
    → /logout → /login

Uses the existing `client` fixture (lifespan-aware TestClient) and the
`fresh_db` autouse fixture (clean users + setup_completed='false' per test).
"""

from __future__ import annotations

import pytest


def _require_phase02_wired() -> None:
    try:
        from app.routers.auth import router  # noqa: F401
        from app.routers.admin import router as admin_router  # noqa: F401
        from app.csrf import CSRFFormFieldShim  # noqa: F401
    except ImportError as exc:
        pytest.skip(f"Wave 4/5 deps not yet present: {exc}")


def test_cold_container_through_login(client, fresh_db) -> None:
    _require_phase02_wired()

    # Step 1: GET /setup renders the form
    r_setup_get = client.get("/setup")
    assert r_setup_get.status_code == 200
    assert "First-time setup" in r_setup_get.text
    token = r_setup_get.cookies.get("csrftoken")
    assert token, "starlette-csrf must set csrftoken cookie on GET /setup response"

    # Step 2: POST /setup with valid form
    r_setup = client.post(
        "/setup",
        data={
            "X-CSRF-Token": token,
            "username": "smoketest",
            "email": "smoke@example.com",
            "password": "twelve-chars-min-password",
        },
        headers={"X-CSRF-Token": token},
        cookies={"csrftoken": token},
        follow_redirects=False,
    )
    assert r_setup.status_code == 303, (
        f"D-03 auto-login: POST /setup must 303 → /, got {r_setup.status_code}: {r_setup.text[:200]}"
    )
    assert r_setup.headers["location"] == "/"
    set_cookie = r_setup.headers.get("set-cookie", "")
    assert "session_id=" in set_cookie, "auto-login must set session_id cookie"

    # Step 3: extract the signed cookie value for the follow-on request
    # (TestClient automatically follows redirects + carries cookies, but
    # follow_redirects=False above means we drive the next step manually.)
    # Parse the session_id value out of the Set-Cookie header:
    import re
    m = re.search(r"session_id=([^;]+)", set_cookie)
    assert m, f"could not extract session_id value: {set_cookie}"
    session_signed = m.group(1)

    # Step 4: GET / with the session cookie shows the auth'd footer
    r_home = client.get(
        "/",
        cookies={"session_id": session_signed, "csrftoken": token},
    )
    assert r_home.status_code == 200
    assert "Signed in as smoketest" in r_home.text, (
        f"home page footer must show 'Signed in as smoketest' for authenticated user; "
        f"got body: {r_home.text[:500]}"
    )
    assert "/logout" in r_home.text, "home page footer must include the Sign-out form action"

    # Step 5: POST /logout
    # Refresh CSRF token from the latest response in case rotation kicked in
    token2 = r_home.cookies.get("csrftoken", token)
    r_logout = client.post(
        "/logout",
        data={"X-CSRF-Token": token2},
        headers={"X-CSRF-Token": token2},
        cookies={"csrftoken": token2, "session_id": session_signed},
        follow_redirects=False,
    )
    assert r_logout.status_code == 303
    assert r_logout.headers["location"] == "/login"

    # Step 6: session cookie cleared
    logout_set_cookie = r_logout.headers.get("set-cookie", "")
    assert "session_id=" in logout_set_cookie
    assert "Max-Age=0" in logout_set_cookie or "max-age=0" in logout_set_cookie, (
        f"logout must emit clear-cookie for session_id (Max-Age=0); got: {logout_set_cookie}"
    )

    # Step 7: GET / after logout shows the "Sign in" link instead of "Signed in as ..."
    # (Don't carry the session_id cookie — it's been cleared client-side too)
    r_anon = client.get("/", cookies={"csrftoken": token2})
    assert r_anon.status_code == 200
    assert "Sign in" in r_anon.text
    assert "Signed in as" not in r_anon.text
```

```
def test_login_failed_no_username_on_user_not_found(client) -> None:
    """D-15: auth.login_failed reason=user_not_found has NO user_id and NO attempted_username."""
    try:
        from app.routers.auth import router  # noqa: F401

    configure_logging(format="json", level="INFO")
    buf, _ = _attach_capture_handler()
    primer = client.get("/")

    if not token:
        pytest.skip("CSRF cookie not primed by GET /")
    client.post(
        "/login",
        data={"X-CSRF-Token": token, "username": "no-such-user-exists", "password": "x"},
        headers={"X-CSRF-Token": token},
        cookies={"csrftoken": token},

    # Find the auth.login_failed line
    import json
    lines = [json.loads(ln) for ln in buf.getvalue().splitlines() if ln.strip() and "auth.login_failed" in ln]
```
