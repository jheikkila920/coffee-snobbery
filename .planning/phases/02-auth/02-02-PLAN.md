def test_dummy_verify_timing() -> None:
    _require_auth_service()
    from app.services.auth import hash_password, verify_password, dummy_verify
    import time, statistics
    real_hash = hash_password("real-password-for-timing-test")
    def t_real() -> float:
        s = time.perf_counter(); verify_password(real_hash, "wrong"); return time.perf_counter() - s
    def t_dummy() -> float:
        s = time.perf_counter(); dummy_verify("anything"); return time.perf_counter() - s
    real_med = statistics.median([t_real() for _ in range(5)])
    dummy_med = statistics.median([t_dummy() for _ in range(5)])
    ratio = dummy_med / real_med
    assert 0.5 < ratio < 2.0, (
        f"dummy_verify timing must be within 0.5x..2.0x of verify_password — "
        f"got dummy={dummy_med*1000:.1f}ms, real={real_med*1000:.1f}ms, ratio={ratio:.2f}. "
        f"If ratio is ~2x, _DUMMY_HASH is likely being recomputed per call (Pitfall 2)."
    )

Functions (per RESEARCH lines 170-189):
```
def hash_password(plaintext: str) -> str:
    """Return an argon2id-encoded hash for *plaintext*.

    The encoded string is ~95-110 chars (Modular Crypt Format) and is what the
    `users.password_hash` TEXT column stores. Phase 9's rehash-on-login path
    consumes `_ph.check_needs_rehash(hash)` separately.
    """
    return _ph.hash(plaintext)


def verify_password(stored_hash: str, candidate_password: str) -> bool:
    """Return True if *candidate_password* matches *stored_hash*; else False.

    Catches `VerifyMismatchError` (wrong password) and `InvalidHashError`
    (malformed/corrupted column value). Other exceptions propagate so genuine
    bugs surface. argon2's `verify()` is constant-time by design.
    """
    try:
        return _ph.verify(stored_hash, candidate_password)
    except (VerifyMismatchError, InvalidHashError):
        return False


def dummy_verify(candidate_password: str) -> None:
    """Constant-time defense for the user-not-found branch (D-15).

    Runs argon2 verify against the precomputed `_DUMMY_HASH` so the
    wall-clock cost matches a real `verify_password()` failure. Result is
    intentionally discarded — `VerifyMismatchError` is the expected outcome.
    """
    try:
        _ph.verify(_DUMMY_HASH, candidate_password)
    except VerifyMismatchError:
        pass  # expected — defense is the timing, not the result.


__all__ = ["hash_password", "verify_password", "dummy_verify"]
```

Phase 1 D-15 logging policy (CONTEXT carried) is the WHY: the `/login` POST handler (Plan 07) calls `dummy_verify(password)` whenever the username lookup returns None, so the wall-clock time of the failed-login response is indistinguishable between "wrong password" and "no such user". That removes the user-enumeration side channel.

DO NOT export `_ph` or `_DUMMY_HASH` from `__all__`. The leading-underscore convention is the gate. The test introspects them via direct attribute access (`from app.services.auth import _ph`) — that's a sanctioned test-side access, not a public contract.
