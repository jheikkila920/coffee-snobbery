Atomic / single-transaction test — use a SQLAlchemy event listener to record every executed statement and confirm the sequence + a single COMMIT:
```
@pytest.mark.asyncio
async def test_for_update_atomic(fresh_db) -> None:
    _require_setup_service()
    from app.services.setup import create_first_admin
    from app.main import async_session_factory, _async_engine
    from sqlalchemy import event
    statements: list[str] = []
    @event.listens_for(_async_engine.sync_engine, "before_cursor_execute")
    def _record(conn, cursor, statement, parameters, context, executemany):
        statements.append(statement.strip().split()[0].upper() if statement.strip() else "")
    commits: list[int] = []
    @event.listens_for(_async_engine.sync_engine, "commit")
    def _commit(conn):
        commits.append(1)
    try:
        async with async_session_factory() as db:
            await create_first_admin(
                db, username="a", email="a@b.c", plaintext_password="twelve-chars-min"
            )
        # Filter out connection-setup noise
        relevant = [s for s in statements if s in ("SELECT", "INSERT", "UPDATE", "COMMIT")]
        # Sequence: SELECT (FOR UPDATE), INSERT (users), UPDATE (app_settings)
        assert "SELECT" in relevant, f"missing SELECT FOR UPDATE: {relevant}"
        select_idx = relevant.index("SELECT")
        insert_idx = relevant.index("INSERT", select_idx) if "INSERT" in relevant[select_idx:] else -1
        update_idx = relevant.index("UPDATE", insert_idx) if "UPDATE" in relevant[insert_idx:] else -1
        assert insert_idx > select_idx, f"INSERT must come after SELECT FOR UPDATE: {relevant}"
        assert update_idx > insert_idx, f"UPDATE must come after INSERT: {relevant}"
        # Exactly one commit (one transaction)
        assert len(commits) == 1, f"exactly one commit expected, got {len(commits)}: {relevant}"
        # Also assert the SQL contains FOR UPDATE
        # The "before_cursor_execute" event yields the full SQL text; we need a second pass
    finally:
        event.remove(_async_engine.sync_engine, "before_cursor_execute", _record)
        event.remove(_async_engine.sync_engine, "commit", _commit)
```

```
async def create_first_admin(
    db: AsyncSession,
    *,
    username: str,
    email: str,
    plaintext_password: str,
) -> User | None:
    """One-transaction first-admin creation. Returns the new User or None on race.

    SQL flow (all inside one implicit AsyncSession transaction):

    1. ``SELECT * FROM app_settings WHERE key = 'setup_completed' FOR UPDATE`` —
       Postgres row-level exclusive lock. A concurrent second caller blocks on
       this SELECT until the first transaction commits (RESEARCH §Postgres 16
       explicit locking, verified). When unblocked, the second caller sees
       ``value='true'`` and returns None — AUTH-02 race protection.
    2. ``INSERT INTO users (username, email, password_hash, is_admin, is_active)``.
       ``is_admin=True`` and ``is_active=True`` are enforced by the service
       (NOT taken from input) per CONTEXT specifics: the first user is always
       admin; no "I am admin" checkbox on the form.
    3. ``UPDATE app_settings SET value='true' WHERE key='setup_completed'``.
    4. ``COMMIT`` — single commit; releases the row lock; both INSERT and
       UPDATE land atomically.

    ``hash_password`` (argon2id, ~100ms) is called inline; the cost is paid
    while holding the FOR UPDATE lock. At household scale this is acceptable;
    the alternative (hash before SELECT, then FOR UPDATE) widens the race
    window because the first caller could lose to a second caller who
    skipped argon2 cost.

    Returns:
        New :class:`User` row with id populated (via ``flush()`` before
        commit) on success; ``None`` if the race was lost.
    """
    stmt = (
        select(AppSetting)
        .where(AppSetting.key == "setup_completed")
        .with_for_update()
    )
    result = await db.execute(stmt)
    setting = result.scalar_one()  # Phase 0 seed row guaranteed present

    if setting.value == "true":
        # Race lost — another concurrent setup completed before our FOR UPDATE
        # could be granted. Roll back to release the lock immediately.
        await db.rollback()
        return None

    new_user = User(
        username=username,
        email=email,
        password_hash=hash_password(plaintext_password),
        is_admin=True,
        is_active=True,
    )
    db.add(new_user)
    # Flush so new_user.id is populated for the caller's audit-log emit;
    # still inside the same transaction (no commit yet).
    await db.flush()

    await db.execute(
        update(AppSetting)
        .where(AppSetting.key == "setup_completed")
        .values(value="true")
    )

    # Single commit — atomic across INSERT and UPDATE; releases the FOR UPDATE lock.
    await db.commit()
    # The User row remains attached to the session after commit because
    # async_session_factory uses expire_on_commit=False (see app/main.py:96).
    return new_user


__all__ = ["create_first_admin"]
```

DO NOT add a wrapping `async with db.begin():` block — the existing `async_session_factory` is `autocommit=off`, so the implicit transaction starts on first execute and commits/rolls back explicitly. This matches the pattern in `app/services/sessions.py::regenerate_session`. Confirmed by RESEARCH §SQLAlchemy 2.0 line 371.

Note for the executor: if `setting.value` is None (the Phase 0 migration seeded `'false'`, but defensive coding suggests None-handling), treat `None != "true"` as "not completed" and proceed with the insert. The Phase 0 seed value is the literal string `"false"` not None — verify via `psql -c "SELECT value FROM app_settings WHERE key='setup_completed'"` if uncertain.
