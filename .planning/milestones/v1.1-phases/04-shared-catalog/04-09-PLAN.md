Module header comment: "Photo upload client-side Canvas downscale (D-05). Reads selected file, draws to canvas at max edge 2000px, re-encodes JPEG 0.85, substitutes the smaller blob into FormData before HTMX submits. Server still re-encodes (SEC-4); this is bandwidth UX optimization."

Implementation:

```
(function () {
  const MAX_EDGE = 2000;
  const JPEG_QUALITY = 0.85;
  const MIN_SIZE_TO_PROCESS = 500_000; // skip if already small

  async function downscale(file) {
    const url = URL.createObjectURL(file);
    try {
      const img = await new Promise((resolve, reject) => {
        const i = new Image();
        i.onload = () => resolve(i);
        i.onerror = reject;
        i.src = url;
      });
      const { w, h } = scaleToMaxEdge(img.naturalWidth, img.naturalHeight);
      const canvas = document.createElement('canvas');
      canvas.width = w;
      canvas.height = h;
      canvas.getContext('2d').drawImage(img, 0, 0, w, h);
      return await new Promise((resolve, reject) => {
        canvas.toBlob(
          (blob) => (blob ? resolve(blob) : reject(new Error('toBlob null'))),
          'image/jpeg',
          JPEG_QUALITY
        );
      });
    } finally {
      URL.revokeObjectURL(url);
    }
  }

  function scaleToMaxEdge(w, h) {
    if (w <= MAX_EDGE && h <= MAX_EDGE) return { w, h };
    const ratio = w > h ? MAX_EDGE / w : MAX_EDGE / h;
    return { w: Math.round(w * ratio), h: Math.round(h * ratio) };
  }

  function wirePhotoForm(form) {
    if (form.__photoUploadWired) return;
    form.__photoUploadWired = true;
    form.addEventListener('submit', async (evt) => {
      if (form.__photoUploadProcessed) return; // retrigger pass
      evt.preventDefault();
      const fileInput = form.querySelector('input[type="file"][data-photo-input]');
      if (!fileInput || !fileInput.files || fileInput.files.length === 0) {
        form.__photoUploadProcessed = true;
        htmx.trigger(form, 'submit');
        return;
      }
      const file = fileInput.files[0];
      if (file.type.startsWith('image/') && file.size >= MIN_SIZE_TO_PROCESS) {
        try {
          const blob = await downscale(file);
          const newFile = new File([blob], file.name.replace(/\.[^.]+$/, '.jpg'), { type: 'image/jpeg' });
          const dt = new DataTransfer();
          dt.items.add(newFile);
          fileInput.files = dt.files;
        } catch (_err) {
          // Fall through and submit the original; server pipeline handles it.
        }
      }
      form.__photoUploadProcessed = true;
      htmx.trigger(form, 'submit');
    }, { capture: true });
  }

  // Wire existing forms + any HTMX-swapped-in forms.
  document.querySelectorAll('form[data-photo-form]').forEach(wirePhotoForm);
  document.body.addEventListener('htmx:afterSettle', (evt) => {
    const root = evt.target || document;
    if (root.matches && root.matches('form[data-photo-form]')) wirePhotoForm(root);
    root.querySelectorAll('form[data-photo-form]').forEach(wirePhotoForm);
  });
})();
```

(Literal block fixes the API surface only. Executor implements as actual JS. CSP-compliant: no inline handlers, no eval.)

LOCKED HOOK STRATEGY: use a form `submit` event listener (NOT `htmx:configRequest`). Reason: HTMX 2.x serializes the form before `htmx:configRequest` fires AND `htmx:configRequest` is synchronous — awaiting inside it is not supported. The correct pattern: `form.addEventListener('submit', async (evt) => { evt.preventDefault(); /* downscale to Blob */; /* wrap in File */; /* replace fileInput.files via DataTransfer */; htmx.trigger(form, 'submit'); }, { capture: true });` — the capture-phase listener intercepts the first submit, and the `htmx.trigger(form, 'submit')` retrigger fires WITHOUT this listener (use a `processed = true` flag on the form to break the loop). Replace the `htmx:configRequest` block in the literal example with the submit-listener form. The test in Task 3 (`test_upload_photo_valid_jpeg_round_trip`) MUST assert that after submit the resulting saved file size on disk is < 1MB when uploading a >2MB synthetic JPEG (downscale path actually ran).
